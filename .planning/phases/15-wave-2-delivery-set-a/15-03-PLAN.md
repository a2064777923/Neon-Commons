test("board manager keeps the same seat when a six has no legal flying chess move", () => {
  resetManagers();
  const manager = getBoardRoomManager();
  const owner = { id: 501, username: "owner501", displayName: "Owner 501" };
  const guest = { id: 502, username: "guest502", displayName: "Guest 502" };
  const room = manager.createRoom(owner, "flyingchess", {
    visibility: "private",
    maxPlayers: 2,
    turnSeconds: 20
  });
  manager.joinRoom(room.roomNo, guest);
  manager.setReady(room.roomNo, owner.id, true);
  manager.setReady(room.roomNo, guest.id, true);

  room.match.turnSeat = 0;
  room.match.turnPhase = FLYING_CHESS_TURN_PHASES.roll;
  room.match.pieces.forEach((piece) => {
    if (piece.seatIndex === 0) {
      piece.progress = 54;
    }
  });

  const originalRandom = Math.random;
  Math.random = () => 0.9999;

  try {
    manager.applyFlyingChessRoll(room, room.players[0], false);
  } finally {
    Math.random = originalRandom;
  }

  assert.equal(room.match.turnSeat, 0);
  assert.equal(room.match.turnPhase, FLYING_CHESS_TURN_PHASES.roll);
  assert.equal(room.match.extraRoll, true);
  assert.deepEqual(room.match.movablePieceIds, []);

  clearTimeout(room.turnTimer);
  delete global.boardRoomManager;
});
```
